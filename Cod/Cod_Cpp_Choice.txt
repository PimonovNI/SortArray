#include <iostream>
using namespace std;

int main()
{
 const int n = 10;
 int array[n] = { 1, 25, 6, 32, 43, 5, 96, 23, 4, 55 };

 int max;
 int pos;

 for (int i = n - 1; i > 0; i--){
 max = array[i];
 pos = i;
 for (int j = 0; j < i; j++)
 if (array[j]>max)
 {
    pos=j;
    max = array[j];
 }
 array[pos] = array[i];
array[i] = max;
 }

 for (int i = 0; i < n; i++){
 cout << array[i] << '\t';
 }
 cout << endl;
}
